Enunciado:
5) Fazer um programa que copie os dados da região de memó-
ria externa de 2100H até 210FH para a região de memória 
externa começando em 2300H. Inicialize os conteúdos destas
posições de memória com valores quaisquer no início do 
programa;




;PROGRAMA EX4.ASM
RESET EQU 0H
	
		ORG RESET
		MOV R0, #0FH ;FAZER O LOOP, E IR DECREMENTANDO O VALOR DE R0
		MOV P2, #23H
		MOV R1, #00H
		MOV A, #1H ;VALOR DO ACC QUE COLOCAREMOS NAS POSIÇÕES DE MEMORIAS
		MOV DPTR, #2100H ;VALOR DA PRIMEIRA POSIÇÃO DE MEMÓRIA EXTERNA
LOOP:	MOVX @DPTR, A ;ESCREVEMOS O VALOR DO ACUMULADOR (1H) NA POSIÇÃO DE MEMÓRIA GUARDADO EM DPTR (X:2200H) NA PRIMEIRA VOLTA
		INC A ;INCREMENTA VALOR DE A => A=1+1
		INC DPTR ;INCREMENTA VALOR DE DPTR => DPTR = 2201H
		DJNZ R0, LOOP ;até aqui colocamos valores na posição e memoria 2100h até 210fh
		MOV R0, #0FH	;COLOQUEI 0FH EM R0 PARA FAZER UM NOVO LOOP P TRANSFERIR OS VALORES
		MOV DPTR, #2100H ;APONTEI O DPTR PARA ONDE EU VOU TIRAR O DADO
TRANSF:	MOVX A, @DPTR ;COPIEI O VALOR DO DADO NO ENDEREÇO DE DPTR PARA O ACC
		MOVX @R1, A ;COPIEI O VALOR DO ACC PARA POSIÇÃO DE MEMÓRIA DE R1 (2300H) POR CAUSA DA PORTA P2 NO COMEÇO DO CODIGO
		INC R1 ;INCREMENTEI PARA PROXIMAS POSIÇÕES DE MEMÓRIA
		INC DPTR ;INCREMENTEI PARA PEGAR OS VALORES DOS OUTROS DADOS;
		DJNZ R0, TRANSF 
		
FIM:	JMP FIM
		END
